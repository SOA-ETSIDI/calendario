\newcommand{\monyear}[2]{

if (Sunday,Saturday) [black!50]
\holidays;
}

\newcommand{\holidays}[0]{
if (equals=01-01,
    equals=03-17,
    equals=12-25,
    equals=12-26) [orange]
if (between = 2017-03-21 and 2017-03-28) [red]
if (equals=01-02,
    equals=03-18,
    equals=12-27,
    equals=12-28) [green];
}
